#SQL queries using SUM, COUNT, AVG, GROUP BY

USE `e-commerce`;

SELECT * FROM FOOD;

SELECT SUM(F_PRICE), MAX(F_PRICE) FROM FOOD;

SELECT COUNT(F_NAME) FROM FOOD;

SELECT AVG(F_PRICE) FROM FOOD;

SELECT COUNT(F_NAME) FROM FOOD GROUP BY F_NAME;

SELECT AVG(F_PRICE), F_NAME FROM FOOD GROUP BY F_NAME HAVING AVG(F_PRICE)<200;
